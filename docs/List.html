def each
  aux = Node.new
  aux = @head
  tam = @num_elem
  for i in (1..tam)
    yield aux.value
    aux = aux.next
  end

end

def extract_head
  extracted_node = Node.new
  extracted_node = @head
  @head = @head.next
  @head.prev = nil
  extracted_node.next = nil
  @num_elem -= 1
  return extracted_node
end

def insert_head(node)
  insertnode = Node.new(node,@head,nil)
  @head.prev = insertnode
  @head = insertnode
  @num_elem += 1
end

def to_s
  aux="{"
  node = Node.new
  node = @head
  while (node != tale) do
    aux += node.value.name
    aux += ", "
    node = node.next
  end
  aux += node.value.name
  aux += "}"
  return aux
end
